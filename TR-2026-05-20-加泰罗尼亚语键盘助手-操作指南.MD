### 7.3 拼写检查引擎

```swift
import Foundation

final class CatKeySpellChecker {

    private let dictionaryManager: DictionaryManager
    private var hunspellInstances: [Language: OpaquePointer] = [:]

    init(dictionaryManager: DictionaryManager) {
        self.dictionaryManager = dictionaryManager
        preloadDictionaries()
    }

    deinit {
        hunspellInstances.values.forEach { Hunspell_destroy($0) }
    }

    private func preloadDictionaries() {
        for language in [.catalan, .spanish] as [Language] {
            if let dictPath = dictionaryManager.dictionaryPath(for: language),
               let affPath = dictionaryManager.affixPath(for: language) {
                let hunspell = Hunspell_create(affPath, dictPath)
                hunspellInstances[language] = hunspell
            }
        }
    }

    func isCorrect(word: String, language: Language) -> Bool {
        guard let hunspell = hunspellInstances[language] else { return true }
        return Hunspell_spell(hunspell, word) != 0
    }

    func suggest(for word: String, language: Language, maxResults: Int = 3) -> [AutocompleteSuggestion] {
        guard let hunspell = hunspellInstances[language] else { return [] }

        var suggestionList: UnsafeMutablePointer<UnsafeMutablePointer<CChar>?>?
        let count = Hunspell_suggest(hunspell, &suggestionList, word)

        defer {
            if let list = suggestionList {
                Hunspell_free_list(hunspell, list, count)
            }
        }

        var results: [AutocompleteSuggestion] = []
        for i in 0..<min(count, maxResults) {
            if let cString = suggestionList?[i] {
                let suggestion = String(cString: cString)
                let distance = word.levenshteinDistance(to: suggestion)
                let score = calculateScore(
                    editDistance: distance,
                    wordFrequency: getFrequency(suggestion, language: language)
                )
                results.append(AutocompleteSuggestion(
                    text: suggestion,
                    isAutocomplete: true,
                    score: score
                ))
            }
        }

        return results.sorted { $0.score > $1.score }
    }

    private func calculateScore(editDistance: Int, wordFrequency: Int) -> Double {
        let distancePenalty = Double(editDistance) * 0.3
        let frequencyBonus = log(Double(wordFrequency + 1)) * 0.2
        return 1.0 - distancePenalty + frequencyBonus
    }

    private func getFrequency(_ word: String, language: Language) -> Int {
        dictionaryManager.wordFrequency(word, language: language)
    }
}
```

### 7.4 语言检测器

```swift
import Foundation

final class LanguageDetector {

    private let catalanMarkers: Set<Character> = ["à", "è", "ç", "ï", "·", "ò"]
    private let spanishMarkers: Set<Character> = ["ñ", "á", "í", "ó", "ú"]

    func detect(currentWord: String, previousWords: [String]) -> Language {
        if hasMarkerCharacters(currentWord, markers: catalanMarkers) {
            return .catalan
        }
        if hasMarkerCharacters(currentWord, markers: spanishMarkers) {
            return .spanish
        }

        if let previousLanguage = inferFromContext(previousWords) {
            return previousLanguage
        }

        return .catalan
    }

    private func hasMarkerCharacters(_ word: String, markers: Set<Character>) -> Bool {
        word.lowercased().contains { markers.contains($0) }
    }

    private func inferFromContext(_ previousWords: [String]) -> Language? {
        var catalanCount = 0
        var spanishCount = 0

        let catalanOnly: Set<String> = ["els", "les", "dels", "amb", "molt",
            "dia", "bon", "bona", "tot", "tota", "tots", "totes", "això",
            "aquí", "també", "però", "més", "seva", "seu", "nostre"]
        let spanishOnly: Set<String> = ["los", "las", "con", "muy",
            "día", "buen", "buena", "todo", "toda", "todos", "todas", "esto",
            "aquí", "también", "pero", "más", "suya", "suyo", "nuestro"]

        for word in previousWords {
            let lower = word.lowercased()
            if catalanOnly.contains(lower) { catalanCount += 2 }
            if spanishOnly.contains(lower) { spanishCount += 2 }
        }

        if catalanCount > spanishCount { return .catalan }
        if spanishCount > catalanCount { return .spanish }
        return nil
    }
}
```

### 7.5 自动补全引擎

```swift
import Foundation

final class CatKeyAutocomplete {

    private let dictionaryManager: DictionaryManager
    private var prefixTrie: PrefixTrie?

    init(dictionaryManager: DictionaryManager) {
        self.dictionaryManager = dictionaryManager
        loadTrie()
    }

    func predict(prefix: String, previousWords: [String], language: Language) -> [AutocompleteSuggestion] {
        var suggestions: [AutocompleteSuggestion] = []

        if let trieMatches = prefixTrie?.search(prefix: prefix.lowercased(), maxResults: 10) {
            for match in trieMatches {
                let score = calculatePredictionScore(
                    word: match.word,
                    prefix: prefix,
                    previousWords: previousWords,
                    frequency: match.frequency,
                    language: language
                )
                suggestions.append(AutocompleteSuggestion(
                    text: match.word,
                    isAutocomplete: true,
                    score: score
                ))
            }
        }

        if let ngramPredictions = predictFromNGram(previousWords: previousWords, language: language) {
            for prediction in ngramPredictions where prediction.hasPrefix(prefix.lowercased()) {
                let score = calculatePredictionScore(
                    word: prediction,
                    prefix: prefix,
                    previousWords: previousWords,
                    frequency: dictionaryManager.wordFrequency(prediction, language: language),
                    language: language
                )
                suggestions.append(AutocompleteSuggestion(
                    text: prediction,
                    isAutocomplete: true,
                    score: score + 0.1
                ))
            }
        }

        return suggestions
            .sorted { $0.score > $1.score }
            .prefix(3)
            .map { $0 }
    }

    private func predictFromNGram(previousWords: [String], language: Language) -> [String]? {
        guard !previousWords.isEmpty else { return nil }
        return dictionaryManager.ngramPredictions(
            for: previousWords.suffix(2).map { $0.lowercased() },
            language: language
        )
    }

    private func calculatePredictionScore(word: String, prefix: String, previousWords: [String], frequency: Int, language: Language) -> Double {
        var score = 0.0
        score += log(Double(frequency + 1)) * 0.3
        if word.hasPrefix(prefix.lowercased()) { score += 0.5 }
        if word.count == prefix.count + 1 { score += 0.2 }
        return score
    }
}
```

### 7.6 编辑距离算法

```swift
extension String {

    func levenshteinDistance(to target: String) -> Int {
        let source = Array(self.lowercased())
        let target = Array(target.lowercased())
        let sourceCount = source.count
        let targetCount = target.count

        guard sourceCount > 0 else { return targetCount }
        guard targetCount > 0 else { return sourceCount }

        var matrix = [[Int]](repeating: [Int](repeating: 0, count: targetCount + 1), count: sourceCount + 1)

        for i in 0...sourceCount { matrix[i][0] = i }
        for j in 0...targetCount { matrix[0][j] = j }

        for i in 1...sourceCount {
            for j in 1...targetCount {
                let cost = source[i - 1] == target[j - 1] ? 0 : 1
                matrix[i][j] = min(
                    matrix[i - 1][j] + 1,
                    matrix[i][j - 1] + 1,
                    matrix[i - 1][j - 1] + cost
                )
            }
        }

        return matrix[sourceCount][targetCount]
    }

    func keyboardAwareDistance(to target: String, layout: KeyboardLayout = .qwerty) -> Int {
        let baseDistance = levenshteinDistance(to: target)
        if baseDistance > 2 { return baseDistance }

        var adjustedDistance = baseDistance
        let sourceChars = Array(self.lowercased())
        let targetChars = Array(target.lowercased())

        for i in 0..<min(sourceChars.count, targetChars.count) where sourceChars[i] != targetChars[i] {
            if layout.areAdjacent(sourceChars[i], targetChars[i]) {
                adjustedDistance = max(0, adjustedDistance - 1)
            }
        }

        return adjustedDistance
    }
}

enum KeyboardLayout {
    case qwerty

    private static let rows: [[Character]] = [
        Array("qwertyuiop"),
        Array("asdfghjkl"),
        Array("zxcvbnm")
    ]

    func areAdjacent(_ a: Character, _ b: Character) -> Bool {
        let posA = Self.position(of: a)
        let posB = Self.position(of: b)
        guard let posA, let posB else { return false }
        let rowDiff = abs(posA.row - posB.row)
        let colDiff = abs(posA.col - posB.col)
        return rowDiff <= 1 && colDiff <= 1 && (rowDiff + colDiff > 0)
    }

    private static func position(of char: Character) -> (row: Int, col: Int)? {
        for (rowIndex, row) in rows.enumerated() {
            if let colIndex = row.firstIndex(of: char) {
                return (rowIndex, colIndex)
            }
        }
        return nil
    }
}
```

### 10.1 PrefixTrie前缀树

```swift
final class PrefixTrie {
    private class TrieNode {
        var children: [Character: TrieNode] = [:]
        var isEndOfWord = false
        var frequency: Int = 0
    }

    private let root = TrieNode()

    func insert(_ word: String, frequency: Int = 1) {
        var node = root
        for char in word.lowercased() {
            if node.children[char] == nil {
                node.children[char] = TrieNode()
            }
            node = node.children[char]!
        }
        node.isEndOfWord = true
        node.frequency = frequency
    }

    func search(prefix: String, maxResults: Int = 10) -> [(word: String, frequency: Int)] {
        var node = root
        for char in prefix.lowercased() {
            guard let next = node.children[char] else { return [] }
            node = next
        }

        var results: [(word: String, frequency: Int)] = []
        collectWords(from: node, prefix: prefix.lowercased(), results: &results, maxResults: maxResults)
        return results.sorted { $0.frequency > $1.frequency }
    }

    private func collectWords(
        from node: TrieNode,
        prefix: String,
        results: inout [(word: String, frequency: Int)],
        maxResults: Int
    ) {
        if results.count >= maxResults { return }
        if node.isEndOfWord {
            results.append((word: prefix, frequency: node.frequency))
        }
        for (char, childNode) in node.children.sorted(by: { $0.value.frequency > $1.value.frequency }) {
            collectWords(from: childNode, prefix: prefix + String(char), results: &results, maxResults: maxResults)
        }
    }

    func buildFromDictionary(_ words: [(word: String, frequency: Int)]) {
        for entry in words {
            insert(entry.word, frequency: entry.frequency)
        }
    }
}
```

### 10.3 NGramModel语言模型

```swift
final class NGramModel {

    private var bigramCounts: [String: [String: Int]] = [:]
    private var trigramCounts: [String: [String: Int]] = [:]
    private var unigramCounts: [String: Int] = [:]
    private var totalUnigrams: Int = 0

    func load(from url: URL) throws {
        let data = try Data(contentsOf: url)
        let model = try JSONDecoder().decode(NGramData.self, from: data)
        self.bigramCounts = model.bigrams
        self.trigramCounts = model.trigrams
        self.unigramCounts = model.unigrams
        self.totalUnigrams = model.totalUnigrams
    }

    func predictNext(previousWords: [String], maxResults: Int = 5) -> [String] {
        if previousWords.count >= 2 {
            let trigramPredictions = predictTrigram(
                word1: previousWords[previousWords.count - 2].lowercased(),
                word2: previousWords[previousWords.count - 1].lowercased(),
                maxResults: maxResults
            )
            if !trigramPredictions.isEmpty { return trigramPredictions }
        }

        if previousWords.count >= 1 {
            let bigramPredictions = predictBigram(
                previousWord: previousWords.last!.lowercased(),
                maxResults: maxResults
            )
            if !bigramPredictions.isEmpty { return bigramPredictions }
        }

        return topUnigrams(maxResults: maxResults)
    }

    private func predictTrigram(word1: String, word2: String, maxResults: Int) -> [String] {
        let key = "\(word1) \(word2)"
        guard let continuations = trigramCounts[key] else { return [] }
        return continuations
            .sorted { $0.value > $1.value }
            .prefix(maxResults)
            .map { $0.key }
    }

    private func predictBigram(previousWord: String, maxResults: Int) -> [String] {
        guard let continuations = bigramCounts[previousWord] else { return [] }
        return continuations
            .sorted { $0.value > $1.value }
            .prefix(maxResults)
            .map { $0.key }
    }

    private func topUnigrams(maxResults: Int) -> [String] {
        unigramCounts
            .sorted { $0.value > $1.value }
            .prefix(maxResults)
            .map { $0.key }
    }

    func probability(of word: String, given previousWords: [String]) -> Double {
        let lowerWord = word.lowercased()
        let lowerPrev = previousWords.map { $0.lowercased() }

        if lowerPrev.count >= 2 {
            let key = "\(lowerPrev[lowerPrev.count - 2]) \(lowerPrev[lowerPrev.count - 1])"
            if let continuations = trigramCounts[key], let count = continuations[lowerWord] {
                let total = continuations.values.reduce(0, +)
                return Double(count) / Double(total)
            }
        }

        if lowerPrev.count >= 1 {
            if let continuations = bigramCounts[lowerPrev.last!], let count = continuations[lowerWord] {
                let total = continuations.values.reduce(0, +)
                return Double(count) / Double(total)
            }
        }

        if let count = unigramCounts[lowerWord] {
            return Double(count) / Double(totalUnigrams)
        }

        return 0.0001
    }
}

struct NGramData: Codable {
    let unigrams: [String: Int]
    let bigrams: [String: [String: Int]]
    let trigrams: [String: [String: Int]]
    let totalUnigrams: Int
}
```

```swift
import CoreData

final class UserDictionaryManager {

    private let coreDataManager: CoreDataManager

    init(coreDataManager: CoreDataManager = .shared) {
        self.coreDataManager = coreDataManager
    }

    func addWord(_ word: String, language: Language) async throws {
        let context = coreDataManager.viewContext
        let entry = UserDictionaryEntry(context: context)
        entry.word = word.lowercased()
        entry.language = language.rawValue
        entry.frequency = 1
        entry.addedAt = Date()
        try context.save()
    }

    func removeWord(_ word: String, language: Language) async throws {
        let context = coreDataManager.viewContext
        let request: NSFetchRequest<UserDictionaryEntry> = UserDictionaryEntry.fetchRequest()
        request.predicate = NSPredicate(
            format: "word == %@ AND language == %@",
            word.lowercased(), language.rawValue
        )
        let results = try context.fetch(request)
        for entry in results {
            context.delete(entry)
        }
        try context.save()
    }

    func incrementFrequency(_ word: String, language: Language) async throws {
        let context = coreDataManager.viewContext
        let request: NSFetchRequest<UserDictionaryEntry> = UserDictionaryEntry.fetchRequest()
        request.predicate = NSPredicate(
            format: "word == %@ AND language == %@",
            word.lowercased(), language.rawValue
        )
        if let entry = try context.fetch(request).first {
            entry.frequency += 1
            try context.save()
        }
    }

    func allWords(for language: Language) async throws -> [String] {
        let context = coreDataManager.viewContext
        let request: NSFetchRequest<UserDictionaryEntry> = UserDictionaryEntry.fetchRequest()
        request.predicate = NSPredicate(format: "language == %@", language.rawValue)
        request.sortDescriptors = [NSSortDescriptor(key: "frequency", ascending: false)]
        return try context.fetch(request).map { $0.word ?? "" }
    }

    func contains(_ word: String, language: Language) -> Bool {
        let context = coreDataManager.viewContext
        let request: NSFetchRequest<UserDictionaryEntry> = UserDictionaryEntry.fetchRequest()
        request.predicate = NSPredicate(
            format: "word == %@ AND language == %@",
            word.lowercased(), language.rawValue
        )
        return (try? context.count(for: request)) ?? 0 > 0
    }
}
```
